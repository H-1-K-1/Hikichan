RewriteEngine On

# Serve existing files and directories directly
RewriteCond %{REQUEST_FILENAME} -f [OR]
RewriteCond %{REQUEST_FILENAME} -d
RewriteRule ^ - [L]

# Pass missing files through smart_build.php
ErrorDocument 404 /hikitest/smart_build.php

# Pass forbidden files and folders through smart_build.php
ErrorDocument 403 /hikitest/smart_build.php

# Disable directory listing
Options -Indexes
